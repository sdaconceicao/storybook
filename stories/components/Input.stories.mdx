An input is a basic html form element.

    const [value, setValue] = useState(args.value);
    const handleChange = (e) => {
      setValue(e.target.value);
    };
  return (
    <Input
      htmlId="example"
      name="example"
      onChange={(e)=>handleChange('onChange', e)}
      value={value}
      {...args}
    />
  )}}
